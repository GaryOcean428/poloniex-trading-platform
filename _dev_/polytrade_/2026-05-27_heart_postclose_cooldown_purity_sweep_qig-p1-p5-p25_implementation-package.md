def compute_postclose_cooldown_remaining_ms(
    self,
    basin_at_close: Optional[Sequence[float]],
    basin_now: Optional[Sequence[float]],
    last_close_at_ms: float,
    observed_migration_rate_dfr_per_ms: float,
    outcome_severity_factor: float,
    lane_period_ms: float,
    chemistry_recovery_samples_ms: Sequence[float],
    current_time_ms: Optional[float] = None,
) -> int:
    """Compute observer-derived post-close cooldown remaining (ms) or 0.

    P6 + v6.7B §§9.5-9.9 (heart master oscillator, breathing-as-tacking cycle,
    pre-cog, frequency-gravity) + EXP-009 (Anderson decorrelation α=0.089,

      (Anderson α informs decorrelation dynamics where rate observable.)
    - OCEAN: outcome_severity_factor from own_outcome_ring (P&L severity / narrow-path
      severity distribution).

      self derived_tacking_frequency_hz / κ samples) + affective_baseline_recovery_ms
      (self-calibrating from own_chemistry_ring distribution — percentile/MAD or

      literal + env fallback. Returns 0 when all observer signals indicate safe
      (d_FR crossed + phase past + low severity + sufficient self-obs).

    LIVED ONLY 5 enforced: production call-sites (tick/executive/loop post-close),
    hard asserts (finite inputs, ring calibration), negatives (see tests), full

    No Euclidean. All distances Fisher-Rao. Two-channel κ respected via callers.

    assert isinstance(last_close_at_ms, (int, float)) and last_close_at_ms > 0, \
        "compute_postclose_cooldown_remaining_ms: last_close_at_ms must be positive (LIVED ONLY)"
    if current_time_ms is None:
        current_time_ms = time.time() * 1000.0
    assert isinstance(current_time_ms, (int, float)) and current_time_ms > last_close_at_ms, \
        "compute_postclose_cooldown_remaining_ms: current_time_ms must be after close (LIVED ONLY)"

        b_now = np.asarray(basin_now, dtype=float)
        if b_close.size == b_now.size and b_close.size > 0:
            d = fisher_rao_distance(b_close, b_now)  # canonical, QIG-pure
            if d < PI_STRUCT_GRAVITATING_FRACTION and observed_migration_rate_dfr_per_ms > 1e-12:

    # LIVED ONLY provenance assert + return (callers must log the factors)

    return remaining
